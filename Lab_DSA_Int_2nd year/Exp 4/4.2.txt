#include<stdio.h>
#include <math.h>
int i,j,k,l,d;
int count_digits(int num) {
    if (num == 0) {
        return 1;
    }
    else{
        int c = 0;
        while (num > 0) {
            num /= 10;
            c++;
        }
        return c;
    }
}
int max(int a[], int s) {
    int max_val = a[0];
    for (i = 1; i < s; i++) {
        if (a[i] > max_val) {
            max_val = a[i];
        }
    }
    return max_val;
}

void sort(int A[], int s) {
    int b[10][100];
    int bc[10] = {0};
    int max_val = max(A, s);
    int c = count_digits(max_val);

    for (i = 0; i < c; i++) {
        for (j = 0; j < s; j++) {
            d = (A[j] / (int)pow(10, i)) % 10;
            b[d][bc[d]++] = A[j];
        }
        k = 0;
        for (j = 0; j < 10; j++) {
            for (l = 0; l < bc[j]; l++) {
                A[k++] = b[j][l];
            }
            bc[j] = 0;
        }
    }
}

int main() {
    int n;
    printf("Enter size of list:");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d no.s:", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    sort(arr, n);
    printf("\nSorted Array:\n");
    for (i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    return 0;
}
