#include<stdio.h>
void main()
{
    int i, n, s = 0, l=0,h=0,mid=0,chk=0;
    printf("Enter total count of no.s:");
    scanf("%d",&n);
    int a[n];
    printf("Enter %d no.s:", n);
    for (i=0;i<n;i++)
        scanf("%d",&a[i]);
    l=0;
    h=n-1;
    printf("Enter no. to search:");
    scanf("%d",&s);
    for(i=0;i<n;i++)
    {
        mid=(l+h)/2;
        if(s==a[mid])
        {
            chk=1;
            break;
        }
        else if(s>a[mid])
            l = mid+1;
        else if(s<a[mid])
            h = mid - 1;
    }
    if(chk==1)
        printf("Number found");
    else
        printf("Number not found");
}
