#include<stdio.h>
void main()
{
    int i, n, s = 0, chk=0;
    printf("Enter total count of no.s");
    scanf("%d",&n);
    int a[n];
    printf("Enter %d no.s", n);
    for (i=0;i<n;i++)
        scanf("%d",&a[i]);
    printf("Enter no. to search:");
    scanf("%d",&s);
    for(i=0;i<n;i++)
    {
        if(s==a[i])
        {
            chk=1;
            break;
        }
    }
    if(chk==1)
        printf("No. found");
    else
        printf("Number not found");
}
