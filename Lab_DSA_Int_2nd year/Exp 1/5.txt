#include <stdio.h>

int main() {
    int i, n;
    printf("Enter no. of array elements: ");
    scanf("%d", &n);

    int a[n]; 
    printf("Enter %d no.s: ");
    for (i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    
    for (i = 0; i < n; i++) {
        if ((i + 1) % 2 == 0) { 
            a[i] = a[i] * a[i];         }
    }

    printf("Array after squaring even positioned no.s: \n");
    for (i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
    printf("\n"); 

    return 0;
}
