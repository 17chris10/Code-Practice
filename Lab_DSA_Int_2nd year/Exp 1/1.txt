#include <stdio.h>

int main() {
    int i, sum = 0;
    int a[10];
    float avg = 0.0f;

    printf("Enter 10 numbers: ");
    for (i = 0; i < 10; i++) {
        scanf("%d", &a[i]);
        sum = sum + a[i];
    }

    avg = (float)sum / 10.0f;
    printf("Average = %.2f\n", avg);

    return 0;
}
